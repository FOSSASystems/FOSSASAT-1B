45 
46  radio.setDio2AsRfSwitch(true);
47  radio.setWhitening(true, WHITENING_INITIAL);
48

57

60

63 
64  // save current modem

240 void Communication_Process_Packet() {
241  // disable interrupts
242  interruptsEnabled = false;
243 
244  // read data
245  size_t len = radio.getPacketLength();
246  if(len == 0) {
247  dataReceived = false;
248  interruptsEnabled = true;
249  return;
250  }
251  uint8_t frame[MAX_RADIO_BUFFER_LENGTH];
252  int16_t state = radio.readData(frame, len);
253 
254  // check reception state
255  if(state == ERR_NONE) {
256  FOSSASAT_DEBUG_PRINT(F("Frm "));
257  FOSSASAT_DEBUG_PRINTLN(len);
258  FOSSASAT_DEBUG_PRINT_BUFF(frame, len);
259 
260  // check callsign
261  uint8_t callsignLen = Persistent_Storage_Read<uint8_t>(EEPROM_CALLSIGN_LEN_ADDR);
262  char callsign[MAX_STRING_LENGTH + 1];
263  System_Info_Get_Callsign(callsign, callsignLen);
264  if(memcmp(frame, (uint8_t*)callsign, callsignLen - 1) == 0) {
265  // check passed
266  Comunication_Parse_Frame(frame, len);
267  } else {
268  FOSSASAT_DEBUG_PRINTLN(F("Call err!"));
269  Persistent_Storage_Increment_Frame_Counter(false);
270  Communication_Acknowledge(0xFF, 0x01);
271  }
272 
273  } else {
274  FOSSASAT_DEBUG_PRINT(F("Rx fail "));
275  FOSSASAT_DEBUG_PRINT(state);
276  Persistent_Storage_Increment_Frame_Counter(false);
277  Communication_Acknowledge(0xFF, 0x02);
278  }
279 
280  // reset flag
281  dataReceived = false;
282 
283  // enable interrupts
284  interruptsEnabled = true;
285 }

287 void Comunication_Parse_Frame(uint8_t* frame, size_t len) {
288  // read callsign from EEPROM
289  uint8_t callsignLen = Persistent_Storage_Read<uint8_t>(EEPROM_CALLSIGN_LEN_ADDR);
290  char callsign[MAX_STRING_LENGTH + 1];
291  System_Info_Get_Callsign(callsign, callsignLen);
292 
293  // get functionID
294  int16_t functionId = FCP_Get_FunctionID(callsign, frame, len);
295  if(functionId < 0) {
296  FOSSASAT_DEBUG_PRINT(F("FID fail "));
297  FOSSASAT_DEBUG_PRINTLN(functionId);
298  Persistent_Storage_Increment_Frame_Counter(false);
299  Communication_Acknowledge(0xFF, 0x03);
300  return;
301  }
302  FOSSASAT_DEBUG_PRINT(F("FID = "));
303  FOSSASAT_DEBUG_PRINTLN(functionId, HEX);
304 
305  // check encryption
306  int16_t optDataLen = 0;
307  uint8_t optData[MAX_OPT_DATA_LENGTH];
308  if((functionId >= PRIVATE_OFFSET) && (functionId <= (PRIVATE_OFFSET + NUM_PRIVATE_COMMANDS))) {
309  // frame contains encrypted data, decrypt
310  FOSSASAT_DEBUG_PRINTLN(F("Decr"));
311 
312  // get optional data length
313  optDataLen = FCP_Get_OptData_Length(callsign, frame, len, encryptionKey, password);
314  if(optDataLen < 0) {
315  FOSSASAT_DEBUG_PRINT(F("Fail "));
316  FOSSASAT_DEBUG_PRINTLN(optDataLen);
317 
318  // decryption failed, increment invalid frame counter and return
319  Persistent_Storage_Increment_Frame_Counter(false);
320  Communication_Acknowledge(0xFF, 0x04);
321  return;
322  }
323 
324  // get optional data
325  if(optDataLen > 0) {
326  FCP_Get_OptData(callsign, frame, len, optData, encryptionKey, password);
327  }
328 
329  } else if(functionId < PRIVATE_OFFSET) {
330  // no decryption necessary
331 
332  // get optional data length
333  optDataLen = FCP_Get_OptData_Length(callsign, frame, len);
334  if(optDataLen < 0) {
335  // optional data extraction failed,
336  FOSSASAT_DEBUG_PRINT(F("len get fail "));
337  FOSSASAT_DEBUG_PRINTLN(optDataLen);
338 
339  // increment invalid frame counter
340  Persistent_Storage_Increment_Frame_Counter(false);
341  Communication_Acknowledge(0xFF, 0x05);
342  return;
343  }
344 
345  // get optional data
346  if(optDataLen > 0) {
347  FCP_Get_OptData(callsign, frame, len, optData);
348  }
349  } else {
350  // unknown function ID
351  FOSSASAT_DEBUG_PRINT(F("Unk ID "));
352  FOSSASAT_DEBUG_PRINTLN(functionId, HEX);
353  Persistent_Storage_Increment_Frame_Counter(false);
354  Communication_Acknowledge(0xFF, 0x06);
355  return;
356  }
357 
358  // check optional data presence
359  if(optDataLen > 0) {
360  // execute with optional data
361  FOSSASAT_DEBUG_PRINT(F("optLen = "));
362  FOSSASAT_DEBUG_PRINTLN(optDataLen);
363  FOSSASAT_DEBUG_PRINT_BUFF(optData, (uint8_t)optDataLen);
364  Communication_Execute_Function(functionId, optData, optDataLen);
365 
366  } else {
367  // execute without optional data
368  Communication_Execute_Function(functionId);
369  }
370 }

372 void Communication_Execute_Function(uint8_t functionId, uint8_t* optData, size_t optDataLen) {
373  // increment valid frame counter
374  Persistent_Storage_Increment_Frame_Counter(true);
375 
376  // acknowledge frame
377  Communication_Acknowledge(functionId, 0x00);
378 
379  // execute function based on ID
380  switch(functionId) {
381  case CMD_PING:
382  // send pong
383  Communication_Send_Response(RESP_PONG);
384  break;
385 
386  case CMD_RETRANSMIT: {
387  // check message length
388  if (optDataLen <= MAX_STRING_LENGTH) {
389  // respond with the requested data
390  Communication_Send_Response(RESP_REPEATED_MESSAGE, optData, optDataLen);
391  }
392  } break;
393 
394  case CMD_RETRANSMIT_CUSTOM: {
395  // check message length
396  if((optDataLen >= 8) && (optDataLen <= MAX_STRING_LENGTH + 7)) {
397  // change modem configuration
398  int16_t state = Communication_Set_Configuration(optData, optDataLen);
399 
400  // check if the change was successful
401  if(state != ERR_NONE) {
402  FOSSASAT_DEBUG_PRINT(F("Cfg fail "));
403  FOSSASAT_DEBUG_PRINTLN(state);
404  } else {
405  // configuration changed successfully, transmit response
406  Communication_Send_Response(RESP_REPEATED_MESSAGE_CUSTOM, optData + 7, optDataLen - 7, true);
407  }
408  }
409  } break;
410 
411  case CMD_TRANSMIT_SYSTEM_INFO:
412  // send system info via LoRa
413  Communication_Send_System_Info();
414  break;
415 
416  case CMD_GET_PACKET_INFO: {
417  // get last packet info and send it
418  static const uint8_t respOptDataLen = 2*sizeof(uint8_t) + 4*sizeof(uint16_t);
419  uint8_t respOptData[respOptDataLen];
420  uint8_t* respOptDataPtr = respOptData;
421

431  Communication_Frame_Add(&respOptDataPtr, loraValid, "Lv", 1, "");
432 
433  uint16_t loraInvalid = Persistent_Storage_Read<uint16_t>(EEPROM_LORA_INVALID_COUNTER_ADDR);
434  Communication_Frame_Add(&respOptDataPtr, loraInvalid, "Li", 1, "");
435 
436  uint16_t fskValid = Persistent_Storage_Read<uint16_t>(EEPROM_FSK_VALID_COUNTER_ADDR);
437  Communication_Frame_Add(&respOptDataPtr, fskValid, "Fv", 1, "");
438 
439  uint16_t fskInvalid = Persistent_Storage_Read<uint16_t>(EEPROM_FSK_INVALID_COUNTER_ADDR);
440  Communication_Frame_Add(&respOptDataPtr, fskInvalid, "Fi", 1, "");
441 
442  Communication_Send_Response(RESP_PACKET_INFO, respOptData, respOptDataLen);
443  } break;
444 
445  case CMD_GET_STATISTICS: {
446  // check optional data is exactly 1 byte
447  if(Communication_Check_OptDataLen(1, optDataLen)) {
448  // response will have maximum of 34 bytes if all stats are included
449  uint8_t respOptData[34];
450  uint8_t respOptDataLen = 1;
451  uint8_t* respOptDataPtr = respOptData;
452 
453  // copy stat flags
454  uint8_t flags = optData[0];
455  memcpy(respOptDataPtr, &flags, sizeof(uint8_t));
456  respOptDataPtr += sizeof(uint8_t);
457 
458  // get required stats from EEPROM
459  if(flags & 0x01) {
460  // charging voltage
461  Communication_Frame_Add(&respOptDataPtr, Persistent_Storage_Read<uint8_t>(EEPROM_CHARGING_VOLTAGE_STATS_ADDR), "", VOLTAGE_MULTIPLIER, "mV");
462  Communication_Frame_Add(&respOptDataPtr, Persistent_Storage_Read<uint8_t>(EEPROM_CHARGING_VOLTAGE_STATS_ADDR + 1), "", VOLTAGE_MULTIPLIER, "mV");

464  respOptDataLen += 3;

521  }
522

525  }

532 
533  // get deployment counter value and send it
534  uint8_t counter = Persistent_Storage_Read<uint8_t>(EEPROM_DEPLOYMENT_COUNTER_ADDR);
535  Communication_Send_Response(RESP_DEPLOYMENT_STATE, &counter, 1);
536  } break;

539  // restart satellite
540  Pin_Interface_Watchdog_Restart();

542 
543  case CMD_WIPE_EEPROM:
544  // wipe EEPROM and reset all EEPROM variables to default values
545  Persistent_Storage_Wipe();
546  break;
547 
548  case CMD_SET_TRANSMIT_ENABLE: {
549  // check optional data is exactly 1 byte
550  if(Communication_Check_OptDataLen(1, optDataLen)) {
551  // load power config from EEPROM

553 
554  // update transmit enable flag
555  powerConfig.bits.transmitEnabled = optData[0];
556 
557  // save power config from EEPROM
558  Power_Control_Save_Configuration();
559  }
560  } break;
561

563  // check optional data is less than limit
564  if(optDataLen <= MAX_STRING_LENGTH) {
565  // get callsign from frame
566  char newCallsign[MAX_STRING_LENGTH + 1];
567  memcpy(newCallsign, optData, optDataLen);
568  newCallsign[optDataLen] = '\0';
569

571  System_Info_Set_Callsign(newCallsign);
572  FOSSASAT_DEBUG_PRINTLN(newCallsign);
573  }
574  } break;
575

577  // check optional data is exactly 1 byte
578  if(Communication_Check_OptDataLen(1, optDataLen)) {

582  }
583  } break;

593  powerConfig.bits.mpptKeepAliveEnabled = optData[1];

611  }

613 
614  case CMD_SET_RECEIVE_WINDOWS: {
615  // check optional data is exactly 2 bytes
616  if(Communication_Check_OptDataLen(2, optDataLen)) {
617  // set FSK receive length
618  Persistent_Storage_Write<uint8_t>(EEPROM_FSK_RECEIVE_LEN_ADDR, optData[0]);
619 
620  // set LoRa receive length
621  Persistent_Storage_Write<uint8_t>(EEPROM_LORA_RECEIVE_LEN_ADDR, optData[1]);
622 
623  // check if there will be still some receive window open
624  if((Persistent_Storage_Read<uint8_t>(EEPROM_LORA_RECEIVE_LEN_ADDR) == 0) && (Persistent_Storage_Read<uint8_t>(EEPROM_FSK_RECEIVE_LEN_ADDR) == 0)) {

626  Persistent_Storage_Write<uint8_t>(EEPROM_FSK_RECEIVE_LEN_ADDR, FSK_RECEIVE_WINDOW_LENGTH);
627  }
628  }
629  } break;
630 
631  case CMD_RECORD_SOLAR_CELLS: {
632  // check optional data is exactly 3 bytes
633  if(Communication_Check_OptDataLen(3, optDataLen)) {
634  uint16_t numSamples = optData[0];
635 
636  // check number of samples is less than limit
637  if(numSamples > 40) {
638  FOSSASAT_DEBUG_PRINT(F(">40"));
639  break;
640  }
641 
642  // get sample period
643  uint16_t period = 0;
644  memcpy(&period, optData + 1, 2);
645  FOSSASAT_DEBUG_PRINT(F("Rec"));
646 
647  // record all data
648  uint8_t respoOptDataLen = 3 * numSamples;
649  uint8_t respOptData[MAX_OPT_DATA_LENGTH];

651  // check if the battery is good enough to continue
652  #ifdef ENABLE_INTERVAL_CONTROL
653  if(!Power_Control_Check_Battery_Limit()) {
654  // battery check failed, stop measurement and send what we have
655  respoOptDataLen = i;
656  break;
657  }
658  #endif
659 
660  // read voltages
661  respOptData[i] = Pin_Interface_Read_Voltage(ANALOG_IN_SOLAR_A_VOLTAGE_PIN) * (VOLTAGE_UNIT / VOLTAGE_MULTIPLIER);
662  respOptData[i + 1] = Pin_Interface_Read_Voltage(ANALOG_IN_SOLAR_B_VOLTAGE_PIN) * (VOLTAGE_UNIT / VOLTAGE_MULTIPLIER);
663  respOptData[i + 2] = Pin_Interface_Read_Voltage(ANALOG_IN_SOLAR_C_VOLTAGE_PIN) * (VOLTAGE_UNIT / VOLTAGE_MULTIPLIER);
664 
665  // wait for for the next measurement
666  Power_Control_Delay(period * SLEEP_LENGTH_CONSTANT, true, true);
667  }
668 
669  // send response
670  Communication_Send_Response(RESP_RECORDED_SOLAR_CELLS, respOptData, respoOptDataLen);
671  }
672  } break;
673 
674  case CMD_ROUTE:
675  // just transmit the optional data
676  Communication_Transmit(optData, optDataLen);
677  break;
678  }
679 }

702 int16_t Communication_Transmit(uint8_t* data, uint8_t len, bool overrideModem) {
703  // check transmit enable flag
704  #ifdef ENABLE_TRANSMISSION_CONTROL
705  Power_Control_Load_Configuration();
706  if(!powerConfig.bits.transmitEnabled) {
707  FOSSASAT_DEBUG_PRINTLN(F("Tx off cmd"));
708  return(ERR_TX_TIMEOUT);
709  }
710  #endif
711 
712  // disable receive interrupt
713  radio.clearDio1Action();
714 
715  // print frame for debugging
716  FOSSASAT_DEBUG_PRINT(F("Send "));
717  FOSSASAT_DEBUG_PRINTLN(len);
718  FOSSASAT_DEBUG_PRINT_BUFF(data, len);
719 
720  // check if modem should be switched - required for transmissions with custom settings
721  uint8_t modem = currentModem;
722  FOSSASAT_DEBUG_PRINT(F("Use "));
723  if(overrideModem) {
724  FOSSASAT_DEBUG_WRITE(MODEM_LORA);
725  FOSSASAT_DEBUG_PRINTLN(F(" (ovr)"));
726  Communication_Set_Modem(MODEM_LORA);
727  } else {
728  FOSSASAT_DEBUG_WRITE(modem);
729  FOSSASAT_DEBUG_PRINTLN();
730  }
731 
732  // get timeout
733  uint32_t timeout = 0;
734  if(currentModem == MODEM_FSK) {
735  timeout = (float)radio.getTimeOnAir(len) * 5.0;
736  } else {
737  timeout = (float)radio.getTimeOnAir(len) * 1.5;
738  }
739  FOSSASAT_DEBUG_PRINT(F("T/O in: "));
740  FOSSASAT_DEBUG_PRINTLN(timeout);
741 
742  // start transmitting
743  int16_t state = radio.startTransmit(data, len);
744  if(state != ERR_NONE) {
745  FOSSASAT_DEBUG_PRINT(F("Tx fail "));
746  FOSSASAT_DEBUG_PRINTLN(state);
747  return(state);
748  }
749 
750  // wait for transmission finish
751  uint32_t start = micros();
752  uint32_t lastBeat = 0;
753  while(!digitalRead(RADIO_DIO1)) {
754  // pet watchdog every second
755  if(micros() - lastBeat > (uint32_t)WATCHDOG_LOOP_HEARTBEAT_PERIOD * (uint32_t)1000) {
756  Pin_Interface_Watchdog_Heartbeat();
757 
758  // check whether voltage dropped below low power level
759  #ifdef ENABLE_INTERVAL_CONTROL
760  if(powerConfig.bits.lowPowerModeActive) {
761  // we're below low power level, stop the transmission
762  FOSSASAT_DEBUG_PRINTLN(F("Tx off bat"));
763  radio.standby();
764  return(ERR_INVALID_DATA_RATE);
765  }
766  #endif
767 
768  lastBeat = micros();
769  }
770 
771  // check timeout
772  if(micros() - start > timeout) {
773  // timed out while transmitting
774  radio.standby();
775  Communication_Set_Modem(modem);
776  FOSSASAT_DEBUG_PRINTLN(F("Tx t/o"));
777  return(ERR_TX_TIMEOUT);
778  }
779  }
780 
781  // transmission done, set mode standby
782  state = radio.standby();
783 
784  // restore modem
785  if(overrideModem) {
786  Communication_Set_Modem(modem);
787  }
788 
789  // set receive ISR
790  radio.setDio1Action(Communication_Receive_Interrupt);
791 
792  return(state);
793 }
